    <button onclick="fetchData()">Get Data</button>

    <pre id="output"></pre>


    <script>

        async function fetchData() {

            const res = await fetch('/data');

            const data = await res.json();

            document.getElementById("output").textContent = JSON.stringify(data, null, 2);

        }

    </script>
